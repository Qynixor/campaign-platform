from django.http import HttpResponse

def robots_txt(request):
    content = """User-agent: *
Disallow: /admin/
Disallow: /accounts/
Disallow: /tinymce/

Allow: /

Sitemap: https://rallynex.com/sitemap.xml  # Changed: removed 'www.'
"""
    return HttpResponse(content, content_type="text/plain")
